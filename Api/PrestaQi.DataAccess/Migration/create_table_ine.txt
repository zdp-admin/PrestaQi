    INCREMENT 1
    START 1
    MINVALUE 1
    MAXVALUE 9223372036854775807
    CACHE 1;

CREATE TABLE public.ine_account
(
    id integer NOT NULL DEFAULT nextval('ine_account_seq_id'::regclass),
    name character varying,
    last_name character varying,
    curp character varying,
    birth_date character varying,
    address character varying,
    clave_elector character varying,
    path_file_front character varying,
    clave_elector_back character varying,
    path_file_back character varying,
    meta character varying,
    accredited_id integer,
    created_at timestamp without time zone,
    updated_at timestamp without time zone,
    CONSTRAINT ine_account_pkey PRIMARY KEY (id)
)
WITH (
    OIDS = FALSE
);
